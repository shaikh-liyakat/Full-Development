Basic BASH terminal commands




ls       //for listing files

ls -a    //list all files with hidden ones 

ex
./    .git/       DataBase/     package-lock.json  src/
../     Express/   Notes_or_summary.txt  React+Vite/     Tailwind
.vscode/    Node_js/   package.json          Server.js


cp       // for copy the file folder
ex 
cp  file_name

mv      //for move file one directory to another directory
ex

mv server.js C:\Users\ALI\OneDrive\Desktop\server.js


mv      // also use for rename the file or folder

ex

mv server.js C:\Users\ALI\OneDrive\Desktop\new_name.js



touch    // for create file

ex
touch file.txt


mkdir    //  for create directory or folder

ex
mkdir Test_folder 


rm   //for remove files

ex
rm  file.txt


rm -f file.txt   //for remove files forcefully


rmdir  folder        //for remove folder

rm -r  folder   //for remove folder which contain files init


for run multiple commands

mkdir Test_folder | touch file.txt      // this or | symbol
